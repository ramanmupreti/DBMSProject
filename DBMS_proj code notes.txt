find /usr -name "mysql-connector-*.jar" 2>/dev/null  (to find connector .jar file)

javac FillTest.java (compile the java file)

java -cp ".:/usr/lib/java/mysql-connector-j-9.4.0.jar" FillTest  (run it with connector .jar file on classpath in putty)

java -cp ".;[connector .jar file path (no quotes, placed in same folder as .java file prgm)]" EmployeeSearchFrame  (run it with connector .jar file on classpath locally)
